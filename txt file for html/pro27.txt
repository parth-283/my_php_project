<script language="javascript">
document.write("First ten prime numbers are<br>");
var n, flag, i, cnt;
cnt = 1;
n = 1;

while(1)
{
	i=2;
	flag=0;
	
while (i<n)
{
	if(n%i==0)
{
flag=1;
break;
}
i++;
}
if(flag==0)
{
	document.write(n + "<br>");
	cnt++;
}
if(cnt>10)
break;
n++;
}
